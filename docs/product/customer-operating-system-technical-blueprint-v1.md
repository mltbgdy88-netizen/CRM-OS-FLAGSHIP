<!-- PAGE 5 -->

RLS standardı: 
ALTER TABLE customers ENABLE ROW LEVEL SECURITY; 
 
CREATE POLICY tenant_isolation_customers 
ON customers 
USING (tenant_id = current_setting('app.tenant_id')::uuid);

SET app.tenant_id = '<tenant_uuid>'; 
SET app.user_id = '<user_uuid>';
